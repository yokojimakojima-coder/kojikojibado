function repopulateSelects() { const all = getAllPlayers(); // 名簿 const active = getActivePlayers(); // 今日参加チェック済み const byName = new Map(players.map(p => [p.name, p])); // ✅ join候補は「名簿」優先、名簿が空なら「参加チェック」を使う const source = (all && all.length > 0) ? all : active; const joinCandidates = []; source.forEach(name => { const p = byName.get(name); if (!p) { joinCandidates.push({ name, kind: "new" }); } else { // playersにはいるが、今は不参加（抜け）→復帰候補 if (!isAvailableAtRound(name, roundNumber, schedule)) { joinCandidates.push({ name, kind: "return" }); } } }); const joinSel = document.getElementById("joinSelect"); joinSel.innerHTML = ""; if (joinCandidates.length === 0) { const opt = document.createElement("option"); opt.value = ""; opt.textContent = "（途中参加/復帰候補なし）"; joinSel.appendChild(opt); } else { joinCandidates.forEach(obj => { const opt = document.createElement("option"); opt.value = obj.name; opt.textContent = obj.kind === "return" ? `【復帰】${obj.name}` : obj.name; joinSel.appendChild(opt); }); } // 途中抜け候補：今参加中の人 const leaveSel = document.getElementById("leaveSelect"); leaveSel.innerHTML = ""; const leaveCandidates = players .filter(p => isAvailableAtRound(p.name, roundNumber, schedule)) .map(p => p.name); if (leaveCandidates.length === 0) { const opt = document.createElement("option"); opt.value = ""; opt.textContent = "（途中抜け候補なし）"; leaveSel.appendChild(opt); } else { leaveCandidates.forEach(name => { const opt = document.createElement("option"); opt.value = name; opt.textContent = name; leaveSel.appendChild(opt); }); } }
